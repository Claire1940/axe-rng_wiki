<Steps>
  <Step num="1" title="เข้าร่วมชุมชนอย่างเป็นทางการ">
    เข้าร่วมกลุ่ม Roblox ของ Chopsy Games และกด "Like" เกม สิ่งนี้จะมอบ **โบนัส Luck +1** ถาวร ซึ่งจะมีผลกับการสุ่มทุกครั้งที่คุณทำ
  </Step>
  <Step num="2" title="สะสมบัฟจากโพชั่น">

  </Step>
  <Step num="3" title="ลงทุนในการเกิดใหม่ (Rebirth)">

  </Step>
  <Step num="4" title="อัปเกรดทักษะดวง">

  </Step>
</Steps>
