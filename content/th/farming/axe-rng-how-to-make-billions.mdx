  <Step num="2" title="เพิ่มประสิทธิภาพการเก็บเกี่ยว">

  </Step>
  <Step num="3" title="ซิงค์กับการตัด">

  </Step>
  <Step num="4" title="ลงทุนน้ำผึ้งซ้ำ">

  </Step>
</Steps>
